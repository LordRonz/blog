using namespace std;

int main() {
    int n, m;
    scanf("%d %d", &n, &m);
    for(int i = 1; i <= n; ++i) {
        for(int j = 1; j <= m; ++j) {
            if(!(i & 1) && i % 4 == 2 && j == m) {
                printf("#");
            }
            else if(!(i & 1) && i % 4 == 0 && j == 1) {
                printf("#");
            }
            else if(i & 1) {
                printf("#");
            }
            else printf(".");
        }
        printf("\n");
    }
    return 0;
}

//0xACCE97ED;

int main() {
    int t, s;
    scanf("%d", &t);
    while(t--) {
        scanf("%d", &s);
        int res = 0, tmp;
        while(s / 10) {
            tmp = s / 10;
            res += tmp * 10;
            s -= tmp * 10;
            s += tmp;
        }
        res += s;
        printf("%d\n", res);
    }
    return 0;
}

        memset(freq, 0, sizeof(freq));
        for(int i = 0; i < n; ++i) {
            scanf("%d", &a[i]);
            ++freq[a[i]];
        }
        int trgt = 0;
        for(int i = 1; i < 101; ++i) {
            if(freq[i] == 1) {
                trgt = i;
                break;
            }
        }

            if(a[i] == trgt) {
                printf("%d\n", i + 1);
                break;
            }
        }
    }
