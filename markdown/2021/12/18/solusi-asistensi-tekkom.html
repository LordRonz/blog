1. Hephaestus berIQ over 9000

Link

#include <stdio.h>

int main(void)
{
    int n, even = 0, odd = 0, num, eindex, oindex;
    scanf("%d", &n);
    for(int i = 1; i <= n; ++i) {
        scanf("%d", &num);
        if(num & 1) {
            ++odd;
            oindex = i;
        }
        else {
            ++even;
            eindex = i;
        }
    }
    printf("%d\n", odd < even ? oindex : eindex);
    return 0;
}

    scanf("%d", &n);
    while(n--) {
        cin >> a;
        if(mp.find(a) == mp.end()) {
            mp[a] = 1;
        }
        else ++mp[a];
    }
    for(auto& it: mp) {
        if(it.second > res) {
            res = it.second;
            a = it.first;
        }
    }
    cout << a << '\n';
    return 0;
}

        scanf("%lld %lld %lld %lld %lld", &a, &b, &x, &y, &n);

    int t, n, w, cnt[55];
    scanf("%d", &t);
    while(t--) {
        scanf("%d", &n);
        memset(cnt, 0, sizeof(cnt));
        for(int i = 0; i < n; ++i) {
            scanf("%d", &w);
            ++cnt[w];
        }
        int ans = 0;
        for (int s = 2; s <= 2 * n; ++s) {
            int cur = 0;
            for (int i = 1; i < (s + 1) / 2; ++i) {
                if (s - i > n) continue;
                cur += min(cnt[i], cnt[s - i]);
            }
            if (s % 2 == 0) cur += cnt[s / 2] / 2;
            ans = max(ans, cur);
        }
        printf("%d\n", ans);
    }
    return 0;
}
